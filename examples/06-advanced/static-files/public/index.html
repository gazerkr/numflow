Usage Example

const numflow = require('numflow')
const app = numflow()

// Basic usage
app.use(numflow.static('public'))

// Advanced options
app.use('/assets', numflow.static('public', {
  maxAge: '1d',  // 1 day cache
  etag: true,    // Enable ETag
  index: 'index.html'
}))

app.listen(3000)

Configuration Options

Option Type Default Description
maxAge string | number 0 Cache duration (e.g., '1d', '2h', '30m')
etag boolean true Enable ETag generation
index boolean | string true File to serve when accessing a directory
dotfiles 'allow' | 'deny' | 'ignore' 'ignore' Dotfiles access control
